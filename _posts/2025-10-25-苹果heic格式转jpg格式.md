下载并安装：[https://imagemagick.org/archive/binaries/ImageMagick-7.1.2-7-Q16-HDRI-x64-dll.exe](https://imagemagick.org/archive/binaries/ImageMagick-7.1.2-7-Q16-HDRI-x64-dll.exe)


```bat
@echo off
setlocal enabledelayedexpansion
set /p input=拖入文件夹：
cd /d "!input!"
md "原图"
for %%f in (*.heic) do (
    echo 正在转换 %%f
    magick "%%f" "原图\%%~nf.jpg"
)
echo 转换结束
pause
```
